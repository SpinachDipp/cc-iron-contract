<script>
let signaturePad;

window.addEventListener("load", () => {
  const canvas = document.getElementById("customerSignature");
  canvas.width = canvas.offsetWidth;
  canvas.height = 220;
  signaturePad = new SignaturePad(canvas,{penColor:"black"});
  generateJobNumber();
});

// Generate job number
function generateJobNumber(){
  const year = new Date().getFullYear();
  const random = Math.floor(1000 + Math.random()*9000);
  document.getElementById("jobNumber").value = "CCIW-"+year+"-"+random;
}

// Employee mode toggle
function toggleEmployeeMode(){
  const fields = document.querySelectorAll(".employee-only");
  const button = document.getElementById("employeeToggleBtn");

  if(fields[0].style.display === "none" || fields[0].style.display === ""){
    const password = prompt("Enter password for Employee Mode:");
    if(password === "usmc0331"){
      fields.forEach(el=>el.style.display="block");
      button.textContent="Customer Mode";
    } else {
      alert("Incorrect password.");
    }
  } else {
    fields.forEach(el=>el.style.display="none");
    button.textContent="Employee Mode";
  }
}

// Only one checkbox per group
function selectOnlyThis(id, group){
  document.getElementsByName(group).forEach(cb=>{ if(cb.id!==id) cb.checked=false; });
  if(group==="deliveryMethod") toggleFreight();
  if(group==="orderType"){ toggleChangeOrders(); toggleOther(); }
}

// Toggle sections
function toggleFreight(){
  const shipping = document.getElementById("shippingCheckbox");
  document.getElementById("freightCostContainer").style.display = shipping.checked ? "block" : "none";
}
function toggleChangeOrders(){
  const change = document.getElementById("orderChange");
  document.getElementById("changeOrdersSection").style.display = change.checked ? "block" : "none";
}
function toggleOther(){
  const other = document.getElementById("other");
  const container = document.getElementById("otherOrderContainer");
  const input = document.getElementById("otherOrderInput");
  if(other.checked){ container.style.display="block"; input.required=true; }
  else { container.style.display="none"; input.required=false; input.value=""; }
}

// Clear signature
function clearSignature(){ signaturePad.clear(); }

// Format phone
function formatPhone(input){
  let numbers = input.value.replace(/\D/g,'');
  if(numbers.length>10) numbers=numbers.slice(0,10);
  if(numbers.length>6) input.value=numbers.slice(0,3)+"-"+numbers.slice(3,6)+"-"+numbers.slice(6);
  else if(numbers.length>3) input.value=numbers.slice(0,3)+"-"+numbers.slice(3);
  else input.value=numbers;
}

// Validate numbers
function validateNumberInput(evt){
  const char = String.fromCharCode(evt.which);
  if(!/[0-9.]/.test(char)) evt.preventDefault();
}

// Validate form before PDF
function validateForm(){
  const requiredFields = document.querySelectorAll("input[required],textarea[required]");
  for(let f of requiredFields){
    if(!f.value.trim()){ alert("Please fill all required fields."); return false; }
  }

  if(!document.getElementById("pickup").checked &&
     !document.getElementById("localDelivery").checked &&
     !document.getElementById("shippingCheckbox").checked){
       alert("Please select a Delivery Method."); return false;
  }

  if(!document.getElementById("newOrder").checked &&
     !document.getElementById("orderChange").checked &&
     !document.getElementById("repair").checked &&
     !document.getElementById("other").checked){
       alert("Please select an Order Type."); return false;
  }

  if(signaturePad.isEmpty()){ alert("Please sign the contract."); return false; }

  const customer = document.getElementById("customerName").value.trim().toLowerCase();
  const printed = document.getElementById("printedName").value.trim().toLowerCase();
  if(customer !== printed){
    alert("Printed Name under signature must match Customer Name.");
    return false;
  }

  return true;
}

// PDF generation
function generatePDF(){
  if(!validateForm()) return;

  const original = document.getElementById("contractForm");
  const clone = original.cloneNode(true);
  clone.querySelectorAll("input,textarea").forEach(input=>{
    const span = document.createElement("span");
    if(input.type==="checkbox") span.textContent = input.checked?"Yes":"No";
    else span.textContent = input.value;
    input.replaceWith(span);
  });

  // Signature
  const sigCanvas = clone.querySelector("#customerSignature");
  const sigImg = document.createElement("img");
  sigImg.src = signaturePad.toDataURL();
  sigImg.style.width="100%";
  sigImg.style.height="220px";
  if(sigCanvas) sigCanvas.replaceWith(sigImg);

  ["freightCostContainer","changeOrdersSection","otherOrderContainer"].forEach(id=>{
    const el = clone.querySelector("#"+id);
    if(el) el.style.display="block";
  });

  const jobNumber = document.getElementById("jobNumber").value;
  const customerName = document.getElementById("customerName").value.trim().replace(/\s+/g,'_') || "Customer";

  html2pdf().set({
    margin:0.5,
    filename: jobNumber+"_"+customerName+"_Contract.pdf",
    image:{type:'jpeg', quality:0.98},
    html2canvas:{scale:2},
    jsPDF:{unit:'in', format:'letter', orientation:'portrait'}
  }).from(clone).save();
}
